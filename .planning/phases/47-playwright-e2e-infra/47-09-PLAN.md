Rewrite `.github/workflows/ci.yml` as follows. Preserve the existing `quality` job exactly; add a `Test (Vitest)` step after typecheck and before build, and add a new `e2e` job:

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  quality:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4

      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - run: pnpm install --frozen-lockfile

      - name: Lint
        run: pnpm lint

      - name: Type check
        run: pnpm typecheck

      - name: Format check
        run: pnpm format:check

      - name: Test (Vitest)
        run: pnpm test
        env:
          DATABASE_URL: 'postgresql://dummy:dummy@localhost:5432/dummy'
          CLERK_SECRET_KEY: 'sk_test_dummy'
          CLERK_WEBHOOK_SECRET: 'whsec_dummy'
          PLATFORM_ADMIN_SECRET: 'a-dummy-secret-that-is-at-least-sixty-four-characters-long-for-validation-purposes'
          NEXT_PUBLIC_CLERK_PUBLISHABLE_KEY: 'pk_test_dummy'
          NEXT_PUBLIC_APP_URL: 'http://localhost:3000'

      - name: Build
        run: pnpm build
        env:
          DATABASE_URL: 'postgresql://dummy:dummy@localhost:5432/dummy'
          CLERK_SECRET_KEY: 'sk_test_dummy'
          CLERK_WEBHOOK_SECRET: 'whsec_dummy'
          PLATFORM_ADMIN_SECRET: 'a-dummy-secret-that-is-at-least-sixty-four-characters-long-for-validation-purposes'
          NEXT_PUBLIC_CLERK_PUBLISHABLE_KEY: 'pk_test_dummy'
          NEXT_PUBLIC_APP_URL: 'http://localhost:3000'

  e2e:
    runs-on: ubuntu-latest
    needs: quality
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_USER: postgres
          POSTGRES_PASSWORD: postgres
          POSTGRES_DB: nc_e2e
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    env:
      NODE_ENV: test
      E2E_TEST: '1'
      E2E_SEED_ENABLED: '1'
      DATABASE_URL: 'postgresql://postgres:postgres@localhost:5432/nc_e2e'
      CLERK_SECRET_KEY: 'sk_test_dummy'
      CLERK_WEBHOOK_SECRET: 'whsec_dummy'
      PLATFORM_ADMIN_SECRET: 'a-dummy-secret-that-is-at-least-sixty-four-characters-long-for-validation-purposes'
      NEXT_PUBLIC_CLERK_PUBLISHABLE_KEY: 'pk_test_dummy'
      NEXT_PUBLIC_APP_URL: 'http://localhost:3000'

    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4

      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - run: pnpm install --frozen-lockfile

      - name: Get Playwright version
        id: playwright-version
        run: echo "version=$(node -p "require('./node_modules/@playwright/test/package.json').version")" >> $GITHUB_OUTPUT

      - name: Cache Playwright browsers
        uses: actions/cache@v4
        id: playwright-cache
        with:
          path: ~/.cache/ms-playwright
          key: playwright-${{ runner.os }}-${{ steps.playwright-version.outputs.version }}

      - name: Install Playwright chromium
        if: steps.playwright-cache.outputs.cache-hit != 'true'
        run: pnpm exec playwright install --with-deps chromium

      - name: Install chromium OS deps (cache hit path)
        if: steps.playwright-cache.outputs.cache-hit == 'true'
        run: pnpm exec playwright install-deps chromium

      - name: Run drizzle migrations
        run: pnpm db:migrate

      - name: Run Playwright E2E
        run: pnpm test:e2e

      - name: Upload Playwright report on failure
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: playwright-report
          path: |
            playwright-report/
            test-results/
          retention-days: 7
```

Key decisions:
- `needs: quality` — E2E only runs if lint/typecheck/test/build all passed. Saves runner minutes on broken PRs.
- Postgres service DB name is `nc_e2e` — matches the safety guardrail in `e2e/lib/db.ts` (which requires 'e2e' or 'test' in the DB name).
- Cache key is Playwright version only — shared across OS runners of the same version.
- `db:migrate` runs BEFORE playwright so the schema exists when webServer boots. Playwright's globalSetup also calls migrate() but idempotently (drizzle no-ops on already-migrated schemas).
- Upload report ONLY on failure — keeps green-PR bloat down.
- Traces/screenshots/videos are in `test-results/` per the Playwright config (`trace: 'retain-on-failure'`).

Commit, push, open the phase PR, watch both jobs green. If `e2e` is red, download the artifact and debug from traces. Common first-run failures:
- Missing testid in a component — fix in a follow-up commit
- Timing race on a network request — prefer `await expect(...).toBeVisible()` over `page.waitForTimeout`
- DB URL typo — compare the env.DATABASE_URL vs the service's POSTGRES_DB
