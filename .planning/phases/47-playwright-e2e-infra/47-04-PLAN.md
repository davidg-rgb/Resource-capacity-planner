// TEST-ONLY ROUTE — Phase 47-04.
//
// Triple-gated so it is impossible to hit in production:
//   1. MODULE-LEVEL throw if NODE_ENV === 'production' (next build tree-shakes
//      throwing modules, and if it doesn't, the server crashes loudly on first
//      import in prod — either way, not reachable by a client).
//   2. Runtime 404 if E2E_SEED_ENABLED !== '1'.
//   3. src/proxy.ts route matcher still applies. We deliberately do NOT add
//      /api/test/* to the public matcher, so even if (1) and (2) were defeated,
//      Clerk auth would still protect the route in any non-test environment.
//
// A fourth gate — static invariant test — asserts the route is absent from
// the .next/ production build output (tests/invariants/no-test-routes-in-prod.test.ts).
//
// ADR-004: this route exists ONLY for the E2E test tier.

if (process.env.NODE_ENV === 'production') {
  throw new Error('[api/test/seed] test-only route imported in production build');
}

import { NextResponse } from 'next/server';
import { buildSeed } from '../../../../../tests/fixtures/seed';
// Import DB client + schema from the app's existing data layer.
// Use the same client the rest of the API routes use.
// Adjust this import to match the repo's actual DB module path.
import { db } from '@/lib/db'; // or wherever drizzle db instance lives

export const runtime = 'nodejs';
export const dynamic = 'force-dynamic';

export async function POST(request: Request): Promise<Response> {
  if (process.env.E2E_SEED_ENABLED !== '1') {
    return new NextResponse(null, { status: 404 });
  }

  const bundle = buildSeed('e2e');

  // Truncate + insert. Use a transaction. The exact insert order matches
  // the FK topology: departments → people → projects → allocations/actuals →
  // proposals → batches. Truncate CASCADE handles the reverse direction.
  await db.transaction(async (tx) => {
    // Truncate all application tables. Reuse the table-discovery logic from
    // e2e/lib/db.ts if convenient, or hard-code the list here — this route
    // runs in the SERVER process and cannot import from e2e/.
    await tx.execute(/* sql */ `
      TRUNCATE TABLE
        "change_log", "batches", "proposals", "actuals", "allocations",
        "projects", "people", "departments"
      RESTART IDENTITY CASCADE
    `);
    // Bulk insert from bundle. Adjust table/column names to match schema.
    if (bundle.departments.length) await tx.insert(/* departments table */).values(bundle.departments);
    if (bundle.people.length) await tx.insert(/* people table */).values(bundle.people);
    if (bundle.projects.length) await tx.insert(/* projects table */).values(bundle.projects);
    if (bundle.allocations.length) await tx.insert(/* allocations table */).values(bundle.allocations);
    if (bundle.actuals.length) await tx.insert(/* actuals table */).values(bundle.actuals);
    if (bundle.proposals.length) await tx.insert(/* proposals table */).values(bundle.proposals);
    if (bundle.batches.length) await tx.insert(/* batches table */).values(bundle.batches);
  });

  return NextResponse.json({
    ok: true,
    counts: {
      departments: bundle.departments.length,
      people: bundle.people.length,
      projects: bundle.projects.length,
      allocations: bundle.allocations.length,
      actuals: bundle.actuals.length,
      proposals: bundle.proposals.length,
      batches: bundle.batches.length,
    },
  });
}
```

NOTE to executor: The exact drizzle import paths, schema names, and table names depend on the repo's existing data layer — look at any existing `src/app/api/.../route.ts` that does writes to find the canonical imports. The seed bundle's shape (`SeedPerson`, `SeedProject`, etc. from `tests/fixtures/seed.ts`) must be mapped to whatever drizzle column names the real schema uses; fields on the seed may need renaming (e.g. `slug` → whatever the people table uses, or dropped entirely if not stored). If a field in SeedBundle has no column, drop it from the insert. Run the smoke POST in Task 3 to catch mismatches.

If `change_log` is the only additional table not in the bundle, that's fine — TRUNCATE CASCADE handles it and we leave it empty.

import type { Page, APIRequestContext } from '@playwright/test';

/**
 * Reset the E2E database to the deterministic seed bundle.
 * Call in `beforeEach` of every spec that mutates data. Safe to call in
 * read-only specs too — idempotent and fast (<500ms typical).
 *
 * Uses the Playwright API request context so the call goes through the
 * webServer (which has the real drizzle connection). Does NOT hit
 * e2e/lib/db.ts — that's for globalSetup, this is per-spec.
 */
export async function seedDb(request: APIRequestContext): Promise<void> {
  const res = await request.post('/api/test/seed', {});
  if (!res.ok()) {
    const body = await res.text().catch(() => '(no body)');
    throw new Error(`[e2e] seedDb failed: ${res.status()} ${body}`);
  }
}

/** Convenience overload that accepts a Page. */
export async function seedDbForPage(page: Page): Promise<void> {
  return seedDb(page.request);
}

import { describe, it, expect } from 'vitest';
import { readFileSync, existsSync } from 'node:fs';
import { execSync } from 'node:child_process';
import { globSync } from 'glob';

/**
 * INVARIANT: /api/test/seed must NOT appear in any production build artifact.
 *
 * Strategy:
 *   1. If `.next/` does not exist, skip (local dev runs tend not to build).
 *      CI runs `pnpm build` before tests in the `quality` job, so .next/
 *      will exist there.
 *   2. Grep the built server chunks for 'api/test/seed' or the route handler
 *      source marker 'test-only route imported in production'.
 *   3. Either string appearing means the route survived bundling → FAIL.
 *
 * If this test fails, the NODE_ENV=production module-level throw is NOT
 * tree-shaking correctly, and production builds will crash on first import.
 * Either way, the route is a liability — fix the build, don't relax the test.
 */
describe('invariant: /api/test/seed absent from production build', () => {
  it('is not present in .next/ build output', () => {
    if (!existsSync('.next')) {
      console.warn('[invariant] .next/ missing — run `pnpm build` first. Skipping.');
      return;
    }
    const files = globSync('.next/server/**/*.{js,mjs,cjs}');
    const offenders: string[] = [];
    for (const file of files) {
      const contents = readFileSync(file, 'utf8');
      if (contents.includes('api/test/seed') || contents.includes('test-only route imported in production')) {
        offenders.push(file);
      }
    }
    expect(offenders).toEqual([]);
  });
});
```
