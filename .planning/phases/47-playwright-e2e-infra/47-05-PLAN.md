```ts
import type { Page } from '@playwright/test';

// Mirror the app's PersonaKind. Kept as a local type to avoid importing from
// src/ (which would pull in React / Clerk). Verify this matches
// src/features/personas/persona.types.ts at authoring time — if it drifts,
// update both.
export type PersonaKind = 'admin' | 'pm' | 'line-manager' | 'staff' | 'rd';

/**
 * Set the persona in localStorage BEFORE the next navigation.
 * Must be called before `page.goto()` — uses addInitScript which runs on
 * every page load in this context. Safe to call multiple times; the last
 * call wins.
 *
 * Example:
 *   test.beforeEach(async ({ page }) => {
 *     await personaAs(page, 'pm');
 *     await page.goto('/');
 *   });
 */
export async function personaAs(
  page: Page,
  kind: PersonaKind,
  extras: Record<string, unknown> = {},
): Promise<void> {
  await page.addInitScript(
    ([k, e]) => {
      window.localStorage.setItem('nc:persona', JSON.stringify({ kind: k, ...(e as Record<string, unknown>) }));
    },
    [kind, extras] as const,
  );
}
```

```ts
import type { Page } from '@playwright/test';
import type { PersonaKind } from '../fixtures/persona';
import { personaAs } from '../fixtures/persona';

/**
 * Navigate as a persona. Sets persona, navigates, waits for the persona's
 * root test id to be visible — guards against the hydration race documented
 * in 46-RESEARCH §Pitfall 2.
 *
 * Each persona's root test id is documented in the table below. If a root
 * test id does not yet exist in the app, the Wave 2 plan that authors the
 * first spec for that persona adds `data-testid="{persona}-home-root"` to
 * the appropriate layout element as part of that plan's task.
 */
const ROOT_TESTID: Record<PersonaKind, string> = {
  'admin': 'admin-home-root',
  'pm': 'pm-home-root',
  'line-manager': 'line-manager-home-root',
  'staff': 'staff-home-root',
  'rd': 'rd-home-root',
};

export async function gotoAs(
  page: Page,
  kind: PersonaKind,
  path: string = '/',
): Promise<void> {
  await personaAs(page, kind);
  await page.goto(path);
  // Optional: await the root test id if it exists. Non-blocking if missing —
  // individual specs can add their own explicit waits for stronger guarantees.
  await page.getByTestId(ROOT_TESTID[kind]).waitFor({ state: 'visible', timeout: 5_000 }).catch(() => {
    // Root test id may not exist yet; do not fail — specs handle their own waits.
  });
}
```

```ts
import { test as base, expect } from '@playwright/test';
import { seedDb } from '../lib/seed';
import { personaAs, type PersonaKind } from './persona';

/**
 * Extended Playwright test with auto-seeding before every test.
 *
 * Usage:
 *   import { test, expect } from '../fixtures/test-base';
 *   import { personaAs } from '../fixtures/persona';
 *
 *   test('TC-E2E-1A: PM Monday check-in', async ({ page }) => {
 *     await personaAs(page, 'pm');
 *     await page.goto('/');
 *     // ... assertions
 *   });
 *
 * The `seedFirst` fixture is auto-used — every test gets a fresh seeded DB
 * before its body runs. Specs that need to mutate and re-check state can
 * rely on the deterministic bundle from tests/fixtures/seed.ts buildSeed('e2e').
 */
type Fixtures = {
  seedFirst: void;
};

export const test = base.extend<Fixtures>({
  seedFirst: [
    async ({ request }, use) => {
      await seedDb(request);
      await use();
    },
    { auto: true },
  ],
});

export { expect, personaAs };
export type { PersonaKind };
```

Rationale for auto-use seedFirst: every TC-E2E spec needs a clean DB, and forgetting the `beforeEach(seedDb)` is a silent cross-test pollution bug. Making it automatic removes the footgun. Cost: read-only specs pay the seed roundtrip (~200-500ms). Acceptable for 12 serial specs.
