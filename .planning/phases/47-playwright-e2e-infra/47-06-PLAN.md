```ts
test.describe('PM historic edit', () => {
  test('TC-E2E-1D: Anna edits a historic allocation via HistoricEditDialog, change_log row asserted', async ({ page, request }) => {
    await personaAs(page, 'pm');
    await page.goto('/');

    // Find a historic (past-month) allocation cell. The seed covers 2026-01..
    // and the E2E test clock is not frozen, so "historic" vs "future" depends
    // on when the test runs. Use 2026-01 which is the earliest month.
    // Navigate to a view showing Jan 2026 for Anna.
    // (specific navigation depends on actual UI — adjust selectors)

    // Open HistoricEditDialog
    const historicCell = page.locator('[data-testid*="historic"], [data-month="2026-01"]').first();
    await historicCell.click();
    await page.getByRole('button', { name: /edit|redigera/i }).click();

    await expect(page.getByRole('dialog')).toBeVisible();
    await expect(page.getByRole('dialog')).toContainText(/historic|historisk/i);

    // Change the hours value
    const hoursField = page.getByLabel(/hours|timmar/i);
    await hoursField.fill('45');
    await page.getByRole('button', { name: /save|spara/i }).click();

    await expect(page.getByRole('dialog')).toBeHidden({ timeout: 5000 });

    // Assert change_log row via the API (most reliable path — no UI scraping)
    // The change-log API path is known from Phase 43-04.
    const logRes = await request.get('/api/admin/change-log?limit=5');
    expect(logRes.ok()).toBe(true);
    const log = await logRes.json();
    // Expect the most recent row to be an allocation edit
    const latest = log.rows?.[0] ?? log[0];
    expect(latest).toBeTruthy();
    expect(JSON.stringify(latest)).toMatch(/allocation|historic/i);
  });
});
```
