// E2E database helpers. Never imported by production code.
import { execSync } from 'node:child_process';

/**
 * Safety guardrail: refuse to operate against any DATABASE_URL whose database
 * name does not contain 'e2e' or 'test'. This is the LAST line of defense
 * against a developer accidentally pointing E2E at the dev or prod DB.
 */
export function assertE2EDatabase(): string {
  const url = process.env.DATABASE_URL;
  if (!url) throw new Error('[e2e/db] DATABASE_URL is not set');
  // Extract the database name from the URL path component.
  let dbName: string;
  try {
    const parsed = new URL(url);
    dbName = parsed.pathname.replace(/^\//, '');
  } catch {
    throw new Error(`[e2e/db] DATABASE_URL is not a valid URL: ${url}`);
  }
  if (!/e2e|test/i.test(dbName)) {
    throw new Error(
      `[e2e/db] Refusing to run E2E against database '${dbName}'. ` +
      `The DB name must contain 'e2e' or 'test'. Set DATABASE_URL to nc_e2e or similar.`,
    );
  }
  return dbName;
}

/**
 * Run drizzle migrations against the E2E database. Spawns the existing
 * `pnpm db:migrate` script so the migration path is identical to dev/prod.
 */
export function migrate(): void {
  assertE2EDatabase();
  execSync('pnpm db:migrate', { stdio: 'inherit', env: process.env });
}

/**
 * Truncate all application tables, preserving schema and drizzle migration
 * metadata. Used by per-spec beforeEach hooks AND by global-setup after
 * migrate() to guarantee a clean slate.
 *
 * Implementation: issue a single `TRUNCATE TABLE ... RESTART IDENTITY CASCADE`
 * against every user table (schema 'public'), excluding drizzle's own
 * `__drizzle_migrations` table. Uses @neondatabase/serverless or `pg` —
 * check which is in dependencies and pick that client.
 *
 * NOTE: This function runs in the Playwright runner process, not the Next
 * dev server. It opens its own client connection to DATABASE_URL and closes
 * it after the truncate.
 */
export async function reset(): Promise<void> {
  assertE2EDatabase();
  // Use @neondatabase/serverless (already in deps) for consistency with app code.
  const { neon } = await import('@neondatabase/serverless');
  const sql = neon(process.env.DATABASE_URL!);
  // Discover all user tables in the public schema.
  const rows = await sql`
    SELECT tablename FROM pg_tables
    WHERE schemaname = 'public'
      AND tablename NOT LIKE '\\_\\_drizzle%' ESCAPE '\\'
  ` as Array<{ tablename: string }>;
  if (rows.length === 0) return;
  const tableList = rows.map((r) => `"public"."${r.tablename}"`).join(', ');
  // Single statement — atomic, fast, resets sequences.
  await sql(`TRUNCATE TABLE ${tableList} RESTART IDENTITY CASCADE`);
}
```

NOTE on client choice: `@neondatabase/serverless` works against both Neon and a local Postgres so long as DATABASE_URL is a valid postgres:// URL. If it does NOT work against a vanilla local Postgres (historically it has required HTTPS), fall back to spawning `psql` via execSync with a TRUNCATE statement built the same way. Record which path was taken in the SUMMARY.

import { assertE2EDatabase, migrate, reset } from './lib/db';

/**
 * Playwright globalSetup hook — runs once per `pnpm test:e2e` invocation,
 * BEFORE Playwright spins up the webServer. We cannot hit /api/test/seed
 * here (server not up yet); instead we:
 *
 *   1. Assert DATABASE_URL points at an e2e DB (safety)
 *   2. Run drizzle migrations (idempotent — fast if already migrated)
 *   3. Truncate all tables (clean slate)
 *
 * Per-spec seeding happens in each spec's `beforeEach` by POSTing to
 * /api/test/seed (created in Plan 47-04), which has access to the
 * already-running webServer's DB connection.
 */
export default async function globalSetup() {
  const dbName = assertE2EDatabase();
  console.log(`[e2e] globalSetup: using database '${dbName}'`);
  migrate();
  await reset();
  console.log('[e2e] globalSetup: migrate + reset complete');
}
```
