```ts
// src/features/proposals/use-lm-queue-count.ts
'use client';
import { useQuery } from '@tanstack/react-query';

export function useLmQueueCount(departmentId: string | null, enabled: boolean) {
  return useQuery({
    queryKey: ['lm-queue-count', departmentId],
    queryFn: async () => {
      const res = await fetch(`/api/v5/proposals/queue/count?departmentId=${departmentId}`, { credentials: 'include' });
      if (!res.ok) throw new Error(`lm-queue-count ${res.status}`);
      return res.json() as Promise<{ count: number; departmentId: string }>;
    },
    select: (data) => data.count,
    refetchInterval: 60_000,
    enabled: !!departmentId && enabled,
  });
}
```

OvercommitDialog skeleton (per D-09, RESEARCH §RD-02 lines 319-326):
```tsx
'use client';
import { useEffect } from 'react';
import Link from 'next/link';
import { useTranslations } from 'next-intl';
import { useQuery } from '@tanstack/react-query';

export interface OvercommitDialogProps {
  open: boolean;
  onClose: () => void;
  scope: 'department' | 'project';
  scopeId: string;
  monthKey: string;
}

export function OvercommitDialog({ open, onClose, scope, scopeId, monthKey }: OvercommitDialogProps) {
  const t = useTranslations('v5.rd.overcommitDialog');
  const { data } = useQuery({
    queryKey: ['capacity-breakdown', scope, scopeId, monthKey],
    queryFn: async () => {
      const url = `/api/v5/capacity/breakdown?scope=${scope}&scopeId=${encodeURIComponent(scopeId)}&monthKey=${monthKey}`;
      const res = await fetch(url, { credentials: 'include' });
      if (!res.ok) throw new Error(`capacity-breakdown ${res.status}`);
      return res.json() as Promise<{ projects: Array<{ id: string; name: string; plannedHours: number; pctOfOvercommit: number }>;
                                     people: Array<{ id: string; name: string; plannedHours: number; capacityHours: number; deltaHours: number }> }>;
    },
    enabled: open,
  });

  // Keyboard ESC-dismiss
  useEffect(() => {
    if (!open) return;
    const handler = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose(); };
    document.addEventListener('keydown', handler);
    return () => document.removeEventListener('keydown', handler);
  }, [open, onClose]);

  if (!open) return null;

  return (
    <div role="dialog" aria-modal="true" aria-label={t('title')}
         className="fixed inset-0 bg-black/40 flex items-center justify-center z-50">
      <div className="bg-surface-container rounded-lg p-6 max-w-2xl w-full">
        <h2 className="text-lg font-semibold">{t('title')}</h2>

        <section aria-labelledby="projects-heading" className="mt-4">
          <h3 id="projects-heading">{t('projects')}</h3>
          <ul>{data?.projects.map((p) => (
            <li key={p.id}>
              <Link href={`/projects/${p.id}`} data-clicks="true">
                {p.name} — {p.plannedHours}h ({Math.round(p.pctOfOvercommit * 100)}%)
              </Link>
            </li>
          ))}</ul>
        </section>

        <section aria-labelledby="people-heading" className="mt-4">
          <h3 id="people-heading">{t('people')}</h3>
          <ul>{data?.people.map((p) => (
            <li key={p.id}>
              <Link href={`/staff/${p.id}?month=${monthKey}`} data-clicks="true">
                {p.name} — {p.plannedHours}/{p.capacityHours}h (Δ {p.deltaHours}h)
              </Link>
            </li>
          ))}</ul>
        </section>

        <button onClick={onClose} className="mt-4">{t('close')}</button>
      </div>
    </div>
  );
}
```
