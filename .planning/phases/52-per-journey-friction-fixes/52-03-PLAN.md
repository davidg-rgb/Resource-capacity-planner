usePmWishCounts hook (D-02 + Pitfall #10 — client-side filter is OK for v6.0):
```ts
// src/features/proposals/use-pm-wish-counts.ts
'use client';
import { useQuery } from '@tanstack/react-query';

export function usePmWishCounts(clerkUserId: string, enabled: boolean) {
  return useQuery({
    queryKey: ['pm-wish-counts', clerkUserId],
    queryFn: async () => {
      const url = `/api/v5/proposals?proposerId=${encodeURIComponent(clerkUserId)}&status=proposed,rejected`;
      const res = await fetch(url, { credentials: 'include' });
      if (!res.ok) throw new Error(`pm-wish-counts ${res.status}`);
      return res.json() as Promise<{ proposals: Array<{ status: string }> }>;
    },
    select: (data) => ({
      pending: data.proposals.filter((p) => p.status === 'proposed').length,
      rejected: data.proposals.filter((p) => p.status === 'rejected').length,
    }),
    refetchInterval: 60_000,
    enabled,
  });
}
```

PendingWishChip shape:
```tsx
'use client';
import Link from 'next/link';
import { useTranslations } from 'next-intl';
import { useFlags } from '@/features/flags/flag.context';
import { usePersona } from '@/features/personas/persona.context';
import { usePmWishCounts } from '@/features/proposals/use-pm-wish-counts';

export function PendingWishChip() {
  const { uiV6PerJourney } = useFlags();
  const persona = usePersona();
  const t = useTranslations('v5.pm.pendingWishChip');
  const { data } = usePmWishCounts(persona.clerkUserId ?? '', !!(uiV6PerJourney && persona.kind === 'pm'));
  if (!uiV6PerJourney || persona.kind !== 'pm') return null;
  const pending = data?.pending ?? 0;
  const rejected = data?.rejected ?? 0;
  if (pending + rejected === 0) return null;
  const href = rejected > 0 ? '/pm/wishes?tab=rejected' : '/pm/wishes?tab=proposed';
  return (
    <Link href={href} data-clicks="true" aria-label={t('label', { pending, rejected })}
      className="inline-flex items-center gap-1 rounded-full border px-2 py-1 text-xs bg-surface-container">
      {/* chip content: e.g. `Önskemål: 1 avvisat · 2 väntande` */}
      {rejected > 0 ? t('rejected', { count: rejected }) : t('pending', { count: pending })}
    </Link>
  );
}
```
