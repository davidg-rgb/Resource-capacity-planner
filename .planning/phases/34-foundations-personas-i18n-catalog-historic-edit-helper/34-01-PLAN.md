```ts
// 1. If process.env.NC_TEST_NOW is set AND matches /^\d{4}-\d{2}$/, return it (test injection).
```

   ```ts
    * Per-request cached server clock source for historic-edit guards.
    *
    * Resolution order:
    *   1. process.env.NC_TEST_NOW if it matches /^\d{4}-(0[1-9]|1[0-2])$/ (test injection).
    *      Malformed NC_TEST_NOW is IGNORED, not returned — poisoning the clock with
    *      garbage would cascade into silent historic-check bugs.
    *   2. tx.__nowMonthKey if already populated on this transaction (cache hit).
    *   3. SELECT to_char(CURRENT_DATE, 'YYYY-MM') AS month_key via tx.execute, cache, return.
    *
    * Why not Node new Date()? See ARCHITECTURE §6.3 CLOCK SOURCE:
    *   Production MUST use the database clock to avoid Node/DB drift at midnight CET.
    *
    * @param tx — any object with an .execute() method (Drizzle transaction shape).
    *             The helper attaches __nowMonthKey to it as a side effect for caching.
    */
   const MONTH_KEY_RE = /^\d{4}-(0[1-9]|1[0-2])$/;

   type TxLike = { __nowMonthKey?: string; execute: (query: any) => Promise<any> };

   export async function getServerNowMonthKey(tx: TxLike): Promise<string> {
     const envOverride = process.env.NC_TEST_NOW;
     if (envOverride && MONTH_KEY_RE.test(envOverride)) return envOverride;

     if (tx.__nowMonthKey && MONTH_KEY_RE.test(tx.__nowMonthKey)) return tx.__nowMonthKey;

     // Use Drizzle's sql template when available. Import lazily so unit tests with a
     // plain spy tx don't need to mock drizzle-orm. The query text must contain
     // CURRENT_DATE and YYYY-MM so FOUND-V5-06-c can assert it.
     const { sql } = await import("drizzle-orm");
     const result: any = await tx.execute(sql`SELECT to_char(CURRENT_DATE, 'YYYY-MM') AS month_key`);

     // Drizzle's .execute() in Neon returns { rows: [...] } or an array depending on driver.
     // Normalize.
     const row = Array.isArray(result) ? result[0] : result?.rows?.[0];
     const monthKey = row?.month_key ?? row?.monthKey;
     if (typeof monthKey !== 'string' || !MONTH_KEY_RE.test(monthKey)) {
       throw new Error(`getServerNowMonthKey: unexpected DB result shape: ${JSON.stringify(result).slice(0, 200)}`);
     }
     tx.__nowMonthKey = monthKey;
     return monthKey;
   }
   ```
