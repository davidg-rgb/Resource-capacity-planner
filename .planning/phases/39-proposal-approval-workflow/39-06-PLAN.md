    ```typescript
    'use client';
    import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
    import type { ProposalDTO, ProposalStatus } from './proposal.types';

    export function useCreateProposal() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: async (input: {
          personId: string; projectId: string; month: string;
          proposedHours: number; note?: string | null;
        }): Promise<ProposalDTO> => {
          const res = await fetch('/api/v5/proposals', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(input),
          });
          if (!res.ok) {
            const err = await res.json().catch(() => ({ error: 'request_failed' }));
            throw new Error(err.error ?? `HTTP ${res.status}`);
          }
          return res.json();
        },
        onSuccess: (_, vars) => {
          qc.invalidateQueries({ queryKey: ['proposals'] });
          qc.invalidateQueries({ queryKey: ['proposals', 'person', vars.personId] });
        },
      });
    }

    export function useListProposals(filter: {
      status?: ProposalStatus | ProposalStatus[];
      departmentId?: string;
      proposerId?: string;
      personId?: string;
      projectId?: string;
    }) {
      const qs = new URLSearchParams();
      if (filter.status) qs.set('status', Array.isArray(filter.status) ? filter.status.join(',') : filter.status);
      if (filter.departmentId) qs.set('departmentId', filter.departmentId);
      if (filter.proposerId) qs.set('proposerId', filter.proposerId);
      if (filter.personId) qs.set('personId', filter.personId);
      if (filter.projectId) qs.set('projectId', filter.projectId);

      return useQuery({
        queryKey: ['proposals', filter],
        queryFn: async (): Promise<{ proposals: ProposalDTO[] }> => {
          const res = await fetch(`/api/v5/proposals?${qs.toString()}`);
          if (!res.ok) throw new Error(`HTTP ${res.status}`);
          return res.json();
        },
      });
    }
    ```

       ```tsx
       'use client';
       import { forwardRef, useImperativeHandle, useRef, useState } from 'react';
       import type { CustomCellEditorProps } from 'ag-grid-react';
       import { useCreateProposal } from '@/features/proposals/use-proposals';

       export const ProposalCellEditor = forwardRef(function ProposalCellEditor(
         props: CustomCellEditorProps & {
           context: { personId: string; currentValue: number };
         },
         ref,
       ) {
         const [hours, setHours] = useState<number>(Number(props.value) || 0);
         const [note, setNote] = useState<string>('');
         const createProposal = useCreateProposal();
         const inputRef = useRef<HTMLInputElement>(null);

         useImperativeHandle(ref, () => ({
           getValue: () => props.value,  // DO NOT write through to grid on stopEditing — proposal flow only saves on Submit wish
           isCancelBeforeStart: () => false,
           isCancelAfterEnd: () => true, // always cancel direct write; we submit via the wish button
         }));

         async function handleSubmit() {
           try {
             const month = props.column.getColId();  // 'YYYY-MM'
             const projectId = props.data.projectId;
             await createProposal.mutateAsync({
               personId: props.context.personId,
               projectId,
               month,
               proposedHours: hours,
               note: note.trim() || null,
             });
             props.api.stopEditing(false);
           } catch (err) {
             console.error('Submit wish failed', err);
             // TODO: toast error — use existing toast primitive
           }
         }

         return (
           <div className="bg-surface p-2 rounded border-2 border-dashed border-amber-500 shadow-lg min-w-[240px]">
             <div className="text-xs font-medium mb-1">Submit wish (out-of-department)</div>
             <input
               ref={inputRef}
               type="number"
               min={0}
               max={999.99}
               step={0.25}
               value={hours}
               onChange={(e) => setHours(Number(e.target.value) || 0)}
               className="w-full border rounded px-2 py-1 text-sm"
               autoFocus
             />
             <textarea
               placeholder="Optional note"
               value={note}
               maxLength={1000}
               onChange={(e) => setNote(e.target.value)}
               className="w-full mt-1 border rounded px-2 py-1 text-xs resize-none"
               rows={2}
             />
             <div className="flex gap-2 mt-2">
               <button
                 type="button"
                 onClick={handleSubmit}
                 disabled={createProposal.isPending}
                 className="flex-1 bg-primary text-primary-foreground rounded px-2 py-1 text-xs"
               >
                 {createProposal.isPending ? '…' : 'Submit wish'}
               </button>
               <button
                 type="button"
                 onClick={() => props.api.stopEditing(true)}
                 className="rounded px-2 py-1 text-xs border"
               >
                 Cancel
               </button>
             </div>
           </div>
         );
       });
       ```

       ```tsx
       'use client';
       import { useState } from 'react';
       import { useCreateProposal } from '../use-proposals';

       interface ProposalCellProps {
         personId: string;
         projectId: string;
         month: string;        // 'YYYY-MM'
         initialHours: number;
         onSubmitted?: () => void;
         onCancelled?: () => void;
       }

       export function ProposalCell(props: ProposalCellProps) {
         const [hours, setHours] = useState(props.initialHours);
         const [note, setNote] = useState('');
         const createProposal = useCreateProposal();

         async function handleSubmit() {
           await createProposal.mutateAsync({
             personId: props.personId,
             projectId: props.projectId,
             month: props.month,
             proposedHours: hours,
             note: note.trim() || null,
           });
           props.onSubmitted?.();
         }

         return (
           <div className="border-2 border-dashed border-amber-500 rounded p-2" data-testid="proposal-cell">
             <input
               type="number" min={0} max={999.99} step={0.25}
               value={hours}
               onChange={(e) => setHours(Number(e.target.value) || 0)}
               aria-label="Proposed hours"
             />
             <textarea
               placeholder="Optional note"
               maxLength={1000}

               onChange={(e) => setNote(e.target.value)}

             />
             <button onClick={handleSubmit} disabled={createProposal.isPending}>

             </button>
             <button onClick={() => props.onCancelled?.()}>Cancel</button>
       ```
