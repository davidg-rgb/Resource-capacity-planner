    export function useResubmitProposal() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: async (input: {
          rejectedProposalId: string;
          proposedHours?: number;
          note?: string | null;
          month?: string;
        }) => {
          const { rejectedProposalId, ...body } = input;
          const res = await fetch(`/api/v5/proposals/${rejectedProposalId}/resubmit`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(body),
          });
          if (!res.ok) {
            const err = await res.json().catch(() => ({ error: 'resubmit_failed' }));
            throw new Error(err.code ?? err.error ?? `HTTP ${res.status}`);
          }
          return res.json();
        },
        onSuccess: () => { qc.invalidateQueries({ queryKey: ['proposals'] }); },
      });
    }

    export function useWithdrawProposal() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: async (input: { proposalId: string }) => {
          const res = await fetch(`/api/v5/proposals/${input.proposalId}/withdraw`, { method: 'POST' });
          if (!res.ok) throw new Error(`HTTP ${res.status}`);
          return res.json();
        },
        onSuccess: () => { qc.invalidateQueries({ queryKey: ['proposals'] }); },
      });
    }

    1. `src/features/proposals/ui/my-wishes-panel.tsx`:
       ```tsx
       'use client';
       import { useState } from 'react';
       import { useListProposals, useResubmitProposal, useWithdrawProposal } from '../use-proposals';
       import type { ProposalDTO, ProposalStatus } from '../proposal.types';
       import { WishCard } from './wish-card';

       type Tab = 'proposed' | 'approved' | 'rejected';

       interface MyWishesPanelProps {
         proposerId: string;
       }

       export function MyWishesPanel({ proposerId }: MyWishesPanelProps) {
         const [tab, setTab] = useState<Tab>('proposed');
         const { data, isLoading } = useListProposals({ status: tab as ProposalStatus, proposerId });
         const resubmit = useResubmitProposal();
         const [editing, setEditing] = useState<ProposalDTO | null>(null);
         const [editHours, setEditHours] = useState(0);
         const [editNote, setEditNote] = useState('');

         function openResubmit(p: ProposalDTO) {
           setEditing(p);
           setEditHours(p.proposedHours);
           setEditNote(p.note ?? '');
         }
         async function confirmResubmit() {
           if (!editing) return;
           await resubmit.mutateAsync({
             rejectedProposalId: editing.id,
             proposedHours: editHours,
             note: editNote || null,
           });
           setEditing(null);
         }

         return (
           <div>
             <div className="flex gap-2 border-b mb-3" role="tablist">
               {(['proposed','approved','rejected'] as Tab[]).map((t) => (
                 <button
                   key={t}
                   role="tab"
                   aria-selected={tab === t}
                   onClick={() => setTab(t)}
                   className={`px-3 py-2 text-sm ${tab === t ? 'border-b-2 border-primary' : ''}`}
                 >
                   {t.charAt(0).toUpperCase() + t.slice(1)}
                 </button>
               ))}
             </div>
             {isLoading && <div>Loading…</div>}
             <div className="space-y-2">
               {(data?.proposals ?? []).map((p) => (
                 <WishCard
                   key={p.id}
                   proposal={p}
                   onResubmit={p.status === 'rejected' ? () => openResubmit(p) : undefined}
                 />
               ))}
               {!isLoading && (data?.proposals ?? []).length === 0 && (
                 <div className="text-sm text-muted-foreground">No {tab} wishes.</div>
               )}
             </div>

             {editing && (
               <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50" role="dialog" aria-modal="true">
                 <div className="bg-surface rounded-lg p-4 max-w-md w-full">
                   <h2 className="text-lg font-semibold">Edit &amp; resubmit</h2>
                   <label className="block text-xs mt-2">Hours</label>
                   <input
                     type="number" min={0} max={999.99} step={0.25}
                     value={editHours}
                     onChange={(e) => setEditHours(Number(e.target.value) || 0)}
                     className="w-full border rounded px-2 py-1"
                     aria-label="Proposed hours"
                   />
                   <label className="block text-xs mt-2">Note</label>
                   <textarea
                     value={editNote}
                     maxLength={1000}
                     onChange={(e) => setEditNote(e.target.value)}
                     className="w-full border rounded px-2 py-1"
                     rows={3}
                     aria-label="Note"
                   />
                   <div className="flex gap-2 justify-end mt-3">
                     <button onClick={() => setEditing(null)} className="border rounded px-3 py-1 text-sm">Cancel</button>
                     <button
                       onClick={confirmResubmit}
                       disabled={resubmit.isPending}
                       className="bg-primary text-primary-foreground rounded px-3 py-1 text-sm"
                     >
                       {resubmit.isPending ? '…' : 'Resubmit'}
                     </button>
                   </div>
                 </div>
               </div>
             )}
           </div>
         );
       }
       ```
