         ```tsx
         const { persona } = usePersona();
         const createProposal = useCreateProposal();
         const [pendingHistoric, setPendingHistoric] = useState<{ hours: number; nextStep: 'direct' | 'proposal' } | null>(null);
         const [showProposalPopover, setShowProposalPopover] = useState<{ hours: number } | null>(null);

         async function handleEdit(nextHours: number) {
           const decision = resolveEditGate({
             persona,
             targetPerson: props.targetPerson,
             month: props.cell.monthKey,
             currentMonth: props.currentMonth,
           });
           if (decision === 'blocked') return;
           if (decision === 'direct') {
             await props.onAllocationPatch({ allocationId: props.cell.allocationId!, hours: nextHours, confirmHistoric: false });
             return;
           }
           if (decision === 'proposal') {
             setShowProposalPopover({ hours: nextHours });
             return;
           }
           if (decision === 'historic-warn-direct') {
             setPendingHistoric({ hours: nextHours, nextStep: 'direct' });
             return;
           }
           if (decision === 'historic-warn-proposal') {
             setPendingHistoric({ hours: nextHours, nextStep: 'proposal' });
             return;
           }
         }
         ```
