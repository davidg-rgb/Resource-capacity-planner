   export interface SystemHealthMetrics {
     dbLatencyMs: number;
     dbConnected: boolean;
     memoryUsageMb: { rss: number; heapUsed: number; heapTotal: number };
     version: string;
     uptime: number; // process uptime seconds

   }

2. Export `async function getSystemHealth(): Promise<SystemHealthMetrics>` that:
   - Measures DB latency by timing `db.execute(sql\`SELECT 1\`)` using `performance.now()` before/after (import `sql` from drizzle-orm, `db` from @/db)
   - Reads `process.memoryUsage()` and converts bytes to MB (rounded to 1 decimal)
   - Reads version from `package.json` using a hardcoded import or `process.env.npm_package_version ?? 'unknown'`
   - Reads `process.uptime()` rounded to integer
   - Wraps DB call in try/catch: on failure set `dbConnected: false`, `dbLatencyMs: -1`
   - Returns `SystemHealthMetrics` object with ISO timestamp

- Uses `handleApiError` for error handling (same pattern as `/api/platform/dashboard/route.ts`)
