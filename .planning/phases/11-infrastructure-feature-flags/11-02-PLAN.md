    ```typescript
      _request: NextRequest,
      { params }: { params: Promise<{ orgId: string }> },
    ) {
      try {
        await requirePlatformAdmin();
        const { orgId } = await params;
        const rows = await db
          .select({ flagName: featureFlags.flagName, enabled: featureFlags.enabled })
          .from(featureFlags)
          .where(eq(featureFlags.organizationId, orgId));
        return NextResponse.json(rows);
      } catch (error) {
        return handleApiError(error);
      }
    }
    ```

    **PATCH handler** — upserts a single flag toggle:
    ```typescript
    const toggleSchema = z.object({
      flagName: z.enum(FLAG_NAMES),
      enabled: z.boolean(),
    });

    export async function PATCH(
      request: NextRequest,
      { params }: { params: Promise<{ orgId: string }> },
    ) {
      try {
        const admin = await requirePlatformAdmin();
        const { orgId } = await params;
        const body = toggleSchema.parse(await request.json());

        await db
          .insert(featureFlags)
          .values({
            organizationId: orgId,
            flagName: body.flagName,
            enabled: body.enabled,
            setByAdminId: admin.adminId,
          })
          .onConflictDoUpdate({
            target: [featureFlags.organizationId, featureFlags.flagName],
            set: {
              enabled: body.enabled,
              setByAdminId: admin.adminId,
              updatedAt: new Date(),
            },
          });

        return NextResponse.json({ success: true });
      } catch (error) {
        return handleApiError(error);
      }
    }
    ```
    - IMPORTANT: Use `onConflictDoUpdate` with composite target `[featureFlags.organizationId, featureFlags.flagName]` to handle the upsert pattern. If Drizzle errors on composite target, fall back to `onConflictDoNothing` + separate UPDATE query.

      ```typescript
      'use client';
      import { usePathname, redirect } from 'next/navigation';
      import { useFlags } from './flag.context';
      import { FLAG_ROUTE_MAP, type FlagName } from './flag.types';
      import { useEffect } from 'react';

      export function FlagGuard({ children }: { children: React.ReactNode }) {
        const pathname = usePathname();
        const flags = useFlags();

        // Check if current path is gated by a disabled flag
        for (const [flagName, routes] of Object.entries(FLAG_ROUTE_MAP)) {
          for (const route of routes) {
            if (pathname.startsWith(route) && !flags[flagName as FlagName]) {
              redirect('/input');
            }
          }
        }

        return <>{children}</>;
      }
      ```
