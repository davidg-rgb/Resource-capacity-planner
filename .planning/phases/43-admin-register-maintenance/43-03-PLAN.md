    Each page is a `'use client'` file following this template:
    ```tsx
    'use client';
    export default function AdminXPage() {
      const { orgRole } = useAuth();
      const isClerkAdmin = orgRole === 'org:admin' || orgRole === 'org:owner';
      if (!isClerkAdmin) return <ForbiddenCard />;

      return (
        <PersonaGate allowed={['admin']}>
          <XRegister />
        </PersonaGate>
      );
    }

    function XRegister() {
      const [includeArchived, setIncludeArchived] = useState(false);
      const [editingRow, setEditingRow] = useState<X | null>(null);
      const [drawerOpen, setDrawerOpen] = useState(false);
      const [banner, setBanner] = useState<string | null>(null);

      const list = useRegisterList('x', { includeArchived });
      const createMut = useCreateRegisterRow('x');
      const updateMut = useUpdateRegisterRow('x');
      const archiveMut = useArchiveRegisterRow('x');

      const onArchive = async (row) => {
        try { await archiveMut.mutateAsync(row.id); setBanner(null); }
        catch (err) {
          if (err instanceof DependentRowsError) setBanner(formatBlockerMessage('x', err.blockers));
          else setBanner(t('v5.admin.register.saveError'));
        }
      };

      return (<>
        <RegisterTable entity="x" columns={X_COLUMNS} rows={list.data ?? []}
          isLoading={list.isLoading} error={list.error}
          includeArchived={includeArchived} onToggleArchived={setIncludeArchived}
          onCreate={() => { setEditingRow(null); setDrawerOpen(true); }}
          onEdit={(row) => { setEditingRow(row); setDrawerOpen(true); }}
          onArchive={onArchive}
          banner={banner} />
        <RegisterDrawer open={drawerOpen} onClose={() => setDrawerOpen(false)}
          title={editingRow ? t('v5.admin.register.edit') : t('v5.admin.register.new')}
          formComponent={XForm}
          defaultValues={editingRow ?? {}}
          isArchived={!!editingRow?.archivedAt}
          submitting={createMut.isPending || updateMut.isPending}
          submitError={null}
          onSubmit={async (values) => {
            if (editingRow) await updateMut.mutateAsync({ id: editingRow.id, data: values });
            else await createMut.mutateAsync(values);
            setDrawerOpen(false);
          }} />
      </>);
    }
    ```
