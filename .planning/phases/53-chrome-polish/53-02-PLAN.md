```ts
// src/features/capacity/capacity.service.ts
export async function getOvercommitCount(orgId: string): Promise<number> {
  // Count people whose planned hours > capacity for any month in [currentMonth, currentMonth+3].
  // Reuse existing capacity aggregation patterns from capacity.read.ts / breakdown route.
  // SQL (conceptual):
  //   SELECT COUNT(DISTINCT person_id) FROM allocations a
  //     JOIN people p ON a.person_id = p.id
  //    WHERE p.organization_id = :orgId
  //      AND a.month_key BETWEEN :currentMonth AND :currentMonth+3
  //    GROUP BY a.person_id, a.month_key
  //   HAVING SUM(a.planned_hours) > work_days_in_month * default_hours_per_day
}
```

**Hook signature** (mirrors `useLmQueueCount`):
```ts
// src/features/proposals/use-rd-overcommit-count.ts
'use client';
import { useQuery } from '@tanstack/react-query';
export interface RdOvercommitCountResponse { count: number; }
export function useRdOvercommitCount(enabled: boolean) {
  return useQuery({
    queryKey: ['rd-overcommit-count'],
    queryFn: async () => {
      const res = await fetch('/api/v5/capacity/overcommit/count', { credentials: 'include' });
      if (!res.ok) throw new Error(`rd-overcommit-count ${res.status}`);
      return (await res.json()) as RdOvercommitCountResponse;
    },
    select: (data) => data.count,
    refetchInterval: 60_000,
    enabled,
  });
}
```
