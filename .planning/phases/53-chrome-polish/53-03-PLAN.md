             innerRadius="50%" outerRadius="80%" paddingAngle={2}>

        <Tooltip />
        <Legend />
      </PieChart>

import type { WidgetProps } from '../widget-registry.types';

function DisciplineBreakdownContent({ config, timeRange, onConfigChange }: WidgetProps) {
  const t = useTranslations('v6.polish.discipline');
  const projectId = (config.projectId as string | undefined) ?? undefined;
  const scope = projectId ? 'project' : 'org';
  const defaultChartType = scope === 'project' ? 'donut' : 'bar'; // D-07
  const chartType = ((config.chartType as string | undefined) ?? defaultChartType) as 'bar' | 'donut';

  // Scope inference â†’ correct hook. enabled gate prevents the inactive hook from firing.
  const orgQuery     = useDisciplineBreakdown(timeRange.from, timeRange.to, { enabled: scope === 'org' });
  const projectQuery = useProjectStaffing(projectId ?? '', timeRange.from, timeRange.to, { enabled: scope === 'project' });

  const rows = useMemo(() => {
    if (scope === 'org') return orgQuery.data ?? [];
    if (!projectQuery.data) return [];
    // D-02: normalize useProjectStaffing {people, months} into DisciplineBreakdownRow[] shape
    // Aggregate allocated hours per discipline across people Ã— months.
    return normalizeProjectStaffing(projectQuery.data);
  }, [scope, orgQuery.data, projectQuery.data]);

  // D-02: per-project with small N â†’ progress-bar list
  if (scope === 'project' && rows.length > 0 && rows.length < 3) {
    return <DisciplineDistribution people={projectQuery.data?.people ?? []} months={projectQuery.data?.months ?? []} />;
  }

  const toggle = (next: 'bar' | 'donut') =>
    onConfigChange?.({ ...config, chartType: next });

  return (
    <div className="space-y-3">
      <div role="tablist" aria-label={t('toggleBar') + ' / ' + t('toggleDonut')} className="flex gap-2">
        <button role="tab" aria-selected={chartType === 'bar'}
                onClick={() => toggle('bar')}
                className={chartType === 'bar' ? 'font-semibold underline' : ''}>
          {t('toggleBar')}
        </button>
        <button role="tab" aria-selected={chartType === 'donut'}
                onClick={() => toggle('donut')}
                className={chartType === 'donut' ? 'font-semibold underline' : ''}>
          {t('toggleDonut')}
        </button>
      </div>
      {chartType === 'bar' ? <DisciplineChart data={rows} /> : <DisciplineDonut data={rows} />}
    </div>
  );
}

-- v6.0 Phase 53 POLISH-03: re-point old widget IDs to the new unified widget
UPDATE dashboard_layouts
SET layout = (
  SELECT jsonb_agg(
    CASE
      WHEN placement->>'widgetId' IN ('discipline-chart','discipline-distribution')
        THEN jsonb_set(placement, '{widgetId}', '"discipline-breakdown"')
      ELSE placement
    END
  )
  FROM jsonb_array_elements(layout) placement
)
WHERE layout::text ~* 'discipline-chart|discipline-distribution';
