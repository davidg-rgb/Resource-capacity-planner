    ```ts
    import { readFileSync, readdirSync } from 'node:fs';
    import { join, sep } from 'node:path';
    import { describe, it, expect } from 'vitest';

    const FORBIDDEN = /throw\s+new\s+Error\s*\(/;
    const ROOTS = ['src/app/api/v5', 'src/features'];

    function* walk(dir: string): Generator<string> {
      let entries;
      try { entries = readdirSync(dir, { withFileTypes: true }); }
      catch { return; }
      for (const e of entries) {
        const p = join(dir, e.name);
        if (e.isDirectory()) yield* walk(p);
        else if (e.isFile() && (e.name.endsWith('.ts') || e.name.endsWith('.tsx'))) yield p;
      }
    }

    describe('TC-INV-ERRTAX: no raw `throw new Error` in v5 API routes or feature services', () => {
      for (const root of ROOTS) {
        for (const file of walk(root)) {
          // Only enforce under features/**/*.service.ts; skip other features files
          if (root === 'src/features' && !file.endsWith('.service.ts')) continue;
          const src = readFileSync(file, 'utf8');
          it(`TC-INV-ERRTAX ${file.split(sep).join('/')} uses AppError, not raw Error`, () => {
            expect(src).not.toMatch(FORBIDDEN);
          });
        }
      }
    });
    ```

    Note: this test is the second layer next to the ESLint guard added in 44-02. It is expected to be RED until 44-02's sweep completes; Wave A plans 01/02/03 land together before Wave B asserts green CI.
